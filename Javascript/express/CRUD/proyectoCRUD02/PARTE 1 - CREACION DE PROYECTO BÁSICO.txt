PARTE 1 - CREACION DE PROYECTO BÁSICO
////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
------------------------------------------------------------------------------------------------------------------------------------------
1 - Creamos archivo de dependencias
npm init;
------------------------------------------------------------------------------------------------------------------------------------------
2- instalamos express y body-parser
npm i express body-parser
------------------------------------------------------------------------------------------------------------------------------------------
3 - Instalamos nodemon
npm i --save-dev nodemon
------------------------------------------------------------------------------------------------------------------------------------------
4 - Instalamos colors
npm i colors
------------------------------------------------------------------------------------------------------------------------------------------
5 - Creamos script nodemon en packege.json
 "scripts": {
    "start": "node index.js",
    "dev": "nodemon index.js"
  },
------------------------------------------------------------------------------------------------------------------------------------------
6 - Creamos index.js y añadimos el siguiente contenido
const express = require('express');
const bodyParser = require('body-parser');
const colors = require('colors');

const app = express();
const PORT = 3000;
------------------------------------------------------------------------------------------------------------------------------------------
7 - Definimos Ruta Raiz
app.get('/', (req, res) => {
    let date = new Date();
    console.log(`home route works ------  ${date}`);
    res.send('Bienvenidos a Express');
});
------------------------------------------------------------------------------------------------------------------------------------------
8 - Iniciamos el server
app.listen(port, () => {
    console.log(`Server on port ${port}`.bgGreen.black);
});
